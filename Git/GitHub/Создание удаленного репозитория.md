```bash
# Добавить удаленный репозиторий на свой ПК
git remote add origin https://github.com/...
# git remote - подкоманда, позволяющая управлять удаленными репозиториями
# add -- добавляет удаленный репозиторий локально на ПК
# origin -- это имя, которое будет присвоено репозиторию
# url -- адрес откуда будет стянут удаленный репозиторий
## После этого локальный и удаленный репозиторий будут связаны

# Связывает лок ветку master с удаленной и отправляет ее в репозиторий
git push -u origin master
# git push -- отправляет лок ветку master в удаленный репозиторий origin
# -u связывает локальную ветку master с удаленной origin/master
# Git знает теперь, что local master <-> origin/master, теперь можно писать
# не git push/pull origin master, а просто git push/pull

# Синхронизируется с удаленным репозиторием и скачивает новые коммиты на локальный ПК
git pull

# Клонирует репозиторий с удаленного репозитория в текущую директорию
git clone https://github.com/...

# Клонирует репозиторий в указанную директорию, которая должна быть пустой либо
# Git сам ее создаст, если такой папки не существует
git clone https://github.com/... my_repository

```
